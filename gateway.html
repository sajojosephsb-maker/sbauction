const socket = io('https://spice-auction-server.onrender.com'); const msg = document.getElementById('msg'); // Real-time server status alerts socket.on('connect', () => { msg.innerText = "✅ System Connected"; msg.style.color = "green"; }); socket.on('connect_error', () => { msg.innerText = "⏳ Waking up server... please wait 30 seconds."; msg.style.color = "orange"; }); function handleLogin() { const btn = document.querySelector('button'); btn.innerText = "Verifying..."; btn.disabled = true; const loginId = document.getElementById('loginId').value; const password = document.getElementById('password').value; socket.emit('attemptLogin', { loginId, password }); } socket.on('loginResponse', (res) => { if(res.success) { // Save user data to LocalStorage so portals can identify the user localStorage.setItem('user', JSON.stringify(res.user)); window.location.href = res.target; } else { const btn = document.querySelector('button'); btn.innerText = "Access Portal"; btn.disabled = false; msg.innerText = res.message; msg.style.color = "red"; } });
