// Inside login.html script function login() { const userId = document.getElementById('uid').value; const password = document.getElementById('pwd').value; // Send login request to server socket.emit('attemptLogin', { userId, password }); } socket.on('loginResponse', (res) => { if(res.success) { localStorage.setItem('user', res.userId); window.location.href = res.targetPage; // e.g. buyer.html } else { alert("Invalid ID or Password"); } });
