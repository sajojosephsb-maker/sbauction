// Function for Admin to create Trader or Company function createAccount(role) { const id = role === 'company' ? document.getElementById('cId').value : document.getElementById('tId').value; const pass = role === 'company' ? document.getElementById('cPass').value : document.getElementById('tPass').value; socket.emit('createAccount', { userId: id, password: pass, role: role }); } // Update UI table with both Traders and Companies socket.on('userListUpdate', (users) => { const body = document.getElementById('user-table-body'); body.innerHTML = users.map(u => ` ${u.userId} ${u.role.toUpperCase()} ${u.status} `).join(''); });
